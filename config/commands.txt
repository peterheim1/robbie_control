# ---------------------------------------------------------------------------
# Stop
# ---------------------------------------------------------------------------
[stop]
phrases: stop, halt, freeze, stop everything, emergency stop
say: stopping
flags: cancel_tasks
ros: empty:/voice/stop
ros: twist:/cmd_vel
ros: float64_array:/drive  values=0,0,0,0

# ---------------------------------------------------------------------------
# Head / look directions
# ---------------------------------------------------------------------------
[look right]
phrases: look right, look to the right, turn your head right
say: looking right
ros: joint_traj:/head_controller/joint_trajectory  head_pan_joint=-1.0 head_tilt_joint=0.0

[look left]
phrases: look left, look to the left, turn your head left
say: looking left
ros: joint_traj:/head_controller/joint_trajectory  head_pan_joint=1.0 head_tilt_joint=0.0

[look up]
phrases: look up, tilt up, tilt your head up
say: looking up
ros: joint_traj:/head_controller/joint_trajectory  head_pan_joint=0.0 head_tilt_joint=0.5

[look down]
phrases: look down, tilt down, tilt your head down
say: looking down
ros: joint_traj:/head_controller/joint_trajectory  head_pan_joint=0.0 head_tilt_joint=-0.5

[look forward]
phrases: look forward, look straight, look ahead, look straight ahead
say: looking forward
ros: joint_traj:/head_controller/joint_trajectory  head_pan_joint=0.0 head_tilt_joint=0.0
